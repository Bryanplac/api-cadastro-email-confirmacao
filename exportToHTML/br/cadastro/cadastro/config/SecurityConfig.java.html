package br.cadastro.cadastro.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.security.config.annotation.web.configurers.AbstractHttpConfigurer;
import org.springframework.http.HttpMethod; // Importação necessária para usar HttpMethod

// Habilita a configuração de segurança web
@Configuration
@EnableWebSecurity
public class SecurityConfig {

    // Define o Bean para o PasswordEncoder (BCrypt)
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }

    // Define a cadeia de filtros de segurança (O CORAÇÃO DA SOLUÇÃO)
    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        http
                // Desabilita o CSRF para simplificar o uso com Thymeleaf (comum em testes)
                .csrf(AbstractHttpConfigurer::disable)
                .authorizeHttpRequests(authorize -> authorize
                        // --- LINHA ADICIONADA PARA LIBERAR O POST DA API REST ---
                        // Libera qualquer requisição POST para /api/cadastro (Postman)
                        .requestMatchers(HttpMethod.POST, "/api/cadastro").permitAll()

                        // TORNA ESTAS URLs PÚBLICAS, SEM REQUERER AUTENTICAÇÃO (Thymeleaf)
                        // ESSENCIAL: libera /confirmar-email, /login, /cadastroUsuario e assets estáticos
                        .requestMatchers("/login", "/cadastroUsuario", "/confirmar-email", "/", "/css/**", "/js/**").permitAll()

                        // Qualquer outra requisição (que não seja as liberadas acima) requer autenticação
                        .anyRequest().authenticated()
                )
                // Configura a página de login customizada
                .formLogin(formLogin -> formLogin
                        .loginPage("/login")
                        .permitAll()
                        .defaultSuccessUrl("/", true) // Redireciona para o dashboard após login
                )
                // Configura o logout
                .logout(logout -> logout
                        .permitAll()
                );

        return http.build();
    }
}
